x <- c(1, 3, 2, 5, 4)
sort(x)

## [1] 1 2 3 4 5

sort(x, decreasing=T)

## [1] 5 4 3 2 1

# Exercise 4  #

###############

is.unsorted(women$height)

## [1] FALSE

is.unsorted(women$weight)

## [1] FALSE

women$bmi <- women$weight  / women$height^2 * 703
is.unsorted(women$bmi)

## [1] TRUE

women <- women[order(women$bmi), sort(names(women))]

df <- df[do.call(order, df), ]
df <- df[do.call(order, -df), ]
